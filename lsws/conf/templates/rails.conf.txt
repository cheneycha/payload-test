chrootmode 0
restrained 1
vhroot $SERVER_ROOT/$VH_NAME/
allowsymbollink 1
enablescript 1
setuidmode 2
configfile $SERVER_ROOT/conf/vhosts/$VH_NAME/vhconf.conf

virtualhostconfig  {
    enablegzip 1
    docroot $VH_ROOT/public/

    errorlog  {
        useserver 1
    }

    context / {
        location $VH_ROOT/
        type rails
        accesscontrol 
        maxconns 5
        adddefaultcharset off
        railsenv 1

        rewrite  {
            rules rewritefile .htaccess
            unknownkeywords rewritefile .htaccess
        }
    }

    index  {
        indexfiles index.html
        autoindex 0
        useserver 0
        autoindexuri /_autoindex/default.php
    }

    hotlinkctrl  {
        allowdirectaccess 1
        suffixes gif, jpeg, jpg
        onlyself 1
        enablehotlinkctrl 0
    }

    accesslog $SERVER_ROOT/logs/$VH_NAME.access.log {
        rollingsize 500M
        keepdays 30
        compressarchive 0
        useserver 0
        logheaders 3
    }

    expires  {
        enableexpires 1
    }

    awstats  {
        workingdir $VH_ROOT/awstats
        awstatsuri /awstats/
        sitedomain localhost
        sitealiases 127.0.0.1 localhost
        updatemode 0
        updateinterval 86400
        updateoffset 0
        securedconn 0
    }

    rewrite  {
        enable 0
        loglevel 0
    }

    accesscontrol  {
        allow *
    }
}
